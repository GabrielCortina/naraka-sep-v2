```typescript
import { createClient } from '@/lib/supabase/server'

export async function DELETE() {
  const supabase = await createClient()

  // 1. Auth check (mesmo padrao do POST)
  const { data: { user }, error: authError } = await supabase.auth.getUser()
  if (authError || !user) return Response.json({ error: 'Nao autorizado' }, { status: 401 })

  const { data: userData } = await supabase
    .from('users').select('role').eq('id', user.id).single()
  if (!userData || !['admin', 'lider'].includes(userData.role))
    return Response.json({ error: 'Sem permissao' }, { status: 403 })

  // 2. Obter ultimo importacao_numero da config
  const { data: config } = await supabase
    .from('config').select('valor').eq('chave', 'ultimo_importacao_numero').single()

  if (!config || config.valor === '0') {
    return Response.json({ error: 'Nenhuma importacao para desfazer' }, { status: 404 })
  }

  const importacao_numero = Number(config.valor)

  // 3. Contar pedidos que serao removidos
  const { count } = await supabase
    .from('pedidos')
    .select('*', { count: 'exact', head: true })
    .eq('importacao_numero', importacao_numero)

  // 4. Deletar progresso e reservas dos pedidos da importacao (FK-safe)
  // Buscar IDs dos pedidos primeiro
  const { data: pedidoIds } = await supabase
    .from('pedidos')
    .select('id')
    .eq('importacao_numero', importacao_numero)

  if (pedidoIds && pedidoIds.length > 0) {
    const ids = pedidoIds.map(p => p.id)
    await supabase.from('progresso').delete().in('pedido_id', ids)
    await supabase.from('reservas').delete().in('pedido_id', ids)
  }

  // 5. Deletar pedidos da importacao
  await supabase
    .from('pedidos')
    .delete()
    .eq('importacao_numero', importacao_numero)

  // 6. Decrementar importacao_numero na config
  await supabase
    .from('config')
    .update({ valor: String(importacao_numero - 1) })
    .eq('chave', 'ultimo_importacao_numero')

  // 7. Deletar atribuicoes orfas (card_key que nao tem mais pedidos)
  // Isso e melhor fazer via cleanup geral, mas por seguranca:
  // Se importacao_numero voltar a 0, pode ser que nao existam mais pedidos

  return Response.json({
    success: true,
    importacao_numero,
    pedidos_removidos: count ?? 0,
  })
}
```
