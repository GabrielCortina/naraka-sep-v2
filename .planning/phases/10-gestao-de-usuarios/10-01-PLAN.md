    ```typescript
    export interface UserFormData {
      nome: string
      pin: string
      pinConfirm: string
      role: string
    }

    export interface ValidationErrors {
      nome?: string
      pin?: string
      pinConfirm?: string
      role?: string
    }

    export interface ValidationResult {
      valid: boolean
      errors: ValidationErrors
    }

    export interface ValidationOptions {
      pinRequired?: boolean // true for create, false for edit
    }

    export function validateUserForm(
      data: UserFormData,
      options: ValidationOptions = { pinRequired: true }
    ): ValidationResult {
      const errors: ValidationErrors = {}

      // Nome validation
      if (!data.nome || data.nome.trim().length === 0) {
        errors.nome = 'Nome obrigatorio'
      }

      // Role validation
      const validRoles = ['admin', 'lider', 'separador', 'fardista']
      if (!data.role || !validRoles.includes(data.role)) {
        errors.role = 'Selecione uma funcao'
      }

      // PIN validation
      const pinProvided = data.pin.length > 0
      if (options.pinRequired && !pinProvided) {
        errors.pin = 'PIN deve ter 4-6 digitos'
      } else if (pinProvided) {
        if (!/^\d{4,6}$/.test(data.pin)) {
          errors.pin = 'PIN deve ter 4-6 digitos'
        } else if (data.pin !== data.pinConfirm) {
          errors.pinConfirm = 'PINs nao coincidem'
        }
      }

      return { valid: Object.keys(errors).length === 0, errors }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> API routes | Untrusted input from browser to /api/users/* endpoints |
| API routes -> Supabase Auth | Privileged admin operations via service role key |
| login form -> session | Deactivated user attempting login |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-01 | Elevation of Privilege | /api/users/* | mitigate | Every API route checks dbUser.role === 'admin' via supabaseAdmin lookup before any write operation |
| T-10-02 | Tampering | POST /api/users | mitigate | Server-side validation: nome non-empty, PIN 4-6 digits numeric, role from enum. Reject malformed input with 400 |
| T-10-03 | Elevation of Privilege | login-form.tsx | mitigate | After signInWithPassword, query ativo from public.users; if false, call signOut() and show error (D-10) |
| T-10-04 | Spoofing | POST /api/users | mitigate | Duplicate email detection via Supabase Auth error handling; return 409 with friendly message |
| T-10-05 | Information Disclosure | GET /api/users | mitigate | Only returns id, nome, role, ativo, created_at. No pin_hash or auth tokens exposed |
| T-10-06 | Tampering | PATCH /api/users/[id] | mitigate | Server-side validation on all mutable fields. PIN validated as 4-6 digits. Role validated against enum |
</threat_model>
