**1. Create `app/api/transformacao/route.ts` -- POST: confirm transformacao item**

Follow exact pattern from `app/api/cards/progress/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { createClient } from '@/lib/supabase/server'
import { supabaseAdmin } from '@/lib/supabase/admin'
```

**POST handler logic:**

Step 1 -- Auth:
```typescript
const supabase = await createClient()
const { data: { user }, error: authError } = await supabase.auth.getUser()
if (authError || !user) return NextResponse.json({ error: 'Nao autenticado' }, { status: 401 })
```

Step 2 -- Role check (admin, lider, separador can confirm; fardista cannot):
```typescript
const { data: dbUser } = await supabaseAdmin.from('users').select('role').eq('id', user.id).single()
if (!dbUser || !['admin', 'lider', 'separador'].includes(dbUser.role)) {
  return NextResponse.json({ error: 'Sem permissao para confirmar transformacao' }, { status: 403 })
}
```

Step 3 -- Parse body:
```typescript
let body: unknown
try { body = await request.json() } catch { return NextResponse.json({ error: 'Body invalido' }, { status: 400 }) }
const { transformacao_id, quantidade } = body as { transformacao_id: unknown; quantidade: unknown }
```

Step 4 -- Validate inputs:
```typescript
if (!transformacao_id || typeof transformacao_id !== 'string') {
  return NextResponse.json({ error: 'transformacao_id invalido' }, { status: 400 })
}
if (typeof quantidade !== 'number' || quantidade < 0 || !Number.isInteger(quantidade)) {
  return NextResponse.json({ error: 'Quantidade invalida' }, { status: 400 })
}
```

Step 5 -- Fetch transformacao row and validate strict quantity (D-06, T-07.1-05 Tampering):
```typescript
// eslint-disable-next-line @typescript-eslint/no-explicit-any
const { data: transformacao, error: fetchError } = await (supabaseAdmin as any)
  .from('transformacoes')
  .select('*')
  .eq('id', transformacao_id)
  .single()

if (fetchError || !transformacao) {
  return NextResponse.json({ error: 'Transformacao nao encontrada' }, { status: 404 })
}

if (transformacao.status === 'concluido') {
  return NextResponse.json({ error: 'Transformacao ja concluida' }, { status: 400 })
}

// STRICT validation (D-06): quantity MUST match exactly
if (quantidade !== transformacao.quantidade) {
  return NextResponse.json({ error: `Quantidade deve ser exatamente ${transformacao.quantidade}` }, { status: 400 })
}
```

Step 6 -- Authorization: separador must be assigned to this card (T-07.1-06):
```typescript
if (dbUser.role === 'separador') {
  if (transformacao.separador_id !== user.id) {
    return NextResponse.json({ error: 'Transformacao nao atribuida a voce' }, { status: 403 })
  }
}
```

Step 7 -- Update status to concluido:
```typescript
// eslint-disable-next-line @typescript-eslint/no-explicit-any
const { error: updateError } = await (supabaseAdmin as any)
  .from('transformacoes')
  .update({ status: 'concluido', concluido_at: new Date().toISOString() })
  .eq('id', transformacao_id)

if (updateError) {
  return NextResponse.json({ error: 'Erro ao confirmar transformacao' }, { status: 500 })
}

return NextResponse.json({ success: true })
```

**2. Create `app/api/transformacao/assign/route.ts` -- POST: assign/reassign separador**

Follow exact pattern from `app/api/cards/assign/route.ts`:

Step 1 -- Auth (same as above)
Step 2 -- Role check: only lider/admin can assign (D-05):
```typescript
if (!dbUser || !['admin', 'lider'].includes(dbUser.role)) {
  return NextResponse.json({ error: 'Apenas lider ou admin pode atribuir' }, { status: 403 })
}
```

Step 3 -- Parse body:
```typescript
const { card_key, numero_transformacao, user_id } = body as {
  card_key: unknown; numero_transformacao: unknown; user_id: unknown
}
```

Step 4 -- Validate:
```typescript
if (!card_key || typeof card_key !== 'string') return NextResponse.json({ error: 'card_key invalido' }, { status: 400 })
if (typeof numero_transformacao !== 'number' || !Number.isInteger(numero_transformacao)) return NextResponse.json({ error: 'numero_transformacao invalido' }, { status: 400 })
if (!user_id || typeof user_id !== 'string') return NextResponse.json({ error: 'user_id invalido' }, { status: 400 })
```

Step 5 -- Verify target user exists and is separador:
```typescript
const { data: targetUser } = await supabaseAdmin.from('users').select('id, nome, role').eq('id', user_id).single()
if (!targetUser || targetUser.role !== 'separador') {
  return NextResponse.json({ error: 'Usuario invalido ou nao e separador' }, { status: 400 })
}
```

Step 6 -- Update ALL transformacoes matching card_key + numero_transformacao (D-09 allows reassignment):
```typescript
// eslint-disable-next-line @typescript-eslint/no-explicit-any
const { error: updateError } = await (supabaseAdmin as any)
  .from('transformacoes')
  .update({
    separador_id: user_id,
    separador_nome: targetUser.nome,
    status: 'atribuido',
  })
  .eq('card_key', card_key)
  .eq('numero_transformacao', numero_transformacao)
  .neq('status', 'concluido') // Don't update already-completed items

if (updateError) {
  return NextResponse.json({ error: 'Erro ao atribuir transformacao' }, { status: 500 })
}

return NextResponse.json({ success: true })
```
