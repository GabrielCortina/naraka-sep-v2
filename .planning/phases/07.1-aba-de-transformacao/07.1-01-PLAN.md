Currently the INSERT does NOT include `numero_transformacao`. Add sequential logic per D-11:

Before the existing INSERT (around line 288), add the sequential number computation:

```typescript
// Compute numero_transformacao (D-11): if current max card is all concluido, increment; else use same
const { data: maxRows } = await supabaseAdmin
  .from('transformacoes')
  .select('numero_transformacao, status')
  .eq('card_key', card_key as string)
  .order('numero_transformacao', { ascending: false })

let nextNumeroTransformacao = 1
if (maxRows && maxRows.length > 0) {
  const currentMax = (maxRows[0] as { numero_transformacao: number }).numero_transformacao
  // Check if ALL items with currentMax are concluido
  const currentMaxItems = maxRows.filter(
    (r: { numero_transformacao: number }) => r.numero_transformacao === currentMax
  )
  const allConcluido = currentMaxItems.every(
    (r: { status: string }) => r.status === 'concluido'
  )
  nextNumeroTransformacao = allConcluido ? currentMax + 1 : currentMax
}
```

Then update the INSERT to include `numero_transformacao: nextNumeroTransformacao`:

```typescript
await (supabaseAdmin as any).from('transformacoes').insert({
```
